# Enable tracking of sign-offs for the merges
export GIT_COMMITTER_NAME="zaraknyu"
export GIT_COMMITTER_EMAIL="razenxizu@gmail.com"

for repo in "${REPOS[@]}"; do
    LOCAL_PATH="${repo%%:*}"
    REMOTE_NAME="${repo##*:}"
    REMOTE_URL="$BASE_URL/$REMOTE_NAME.git"
    
    echo "🔄 Merging upstream updates into $LOCAL_PATH..."
    
    # Check if the subdirectory exists; if not, do an initial subtree add
    if [ ! -d "$LOCAL_PATH" ]; then
        echo "📁 Path doesn't exist. Initializing subtree add for $LOCAL_PATH..."
        git subtree add --prefix="$LOCAL_PATH" "$REMOTE_URL" "$TAG" --squash -m "Merge '$LOCAL_PATH/' from TAG $TAG"
    else
        # Perform the actual subtree merge pull
        git subtree pull --prefix="$LOCAL_PATH" "$REMOTE_URL" "$TAG" --squash -m "Merge '$LOCAL_PATH/' from TAG $TAG"
    fi
    
    if [ $? -eq 0 ]; then
        echo "✅ Successfully merged: $LOCAL_PATH"
    else
        echo "❌ CONFLICT OR FAILURE in: $LOCAL_PATH"
        echo "Fix manual conflicts, commit, and then re-run the loop."
        break
    fi
    echo "--------------------------------------------------------"
done
